sudo yum update -y
sudo yum install -y httpd php php-mysqli mariadb

Start Apache:

sudo systemctl start httpd
sudo systemctl enable httpd

db.php (Database Connection):-

<?php
$host = "your-rds-endpoint";  // e.g. mydb.abc123.us-east-1.rds.amazonaws.com
$user = "admin";              // your RDS username
$pass = "password";           // your RDS password
$db   = "ecommerce";          // database name

$conn = mysqli_connect($host, $user, $pass, $db);

if (!$conn) {
    die("Connection failed: " . mysqli_connect_error());
}
?>

register.php (User Registration):-

<?php
include 'db.php';

if ($_SERVER["REQUEST_METHOD"] == "POST") {
    $name  = $_POST['name'];
    $email = $_POST['email'];
    $pass  = password_hash($_POST['password'], PASSWORD_BCRYPT);

    $sql = "INSERT INTO users (name, email, password) VALUES ('$name', '$email', '$pass')";
    
    if (mysqli_query($conn, $sql)) {
        echo "Registration successful! <a href='index.html'>Login</a>";
    } else {
        echo "Error: " . mysqli_error($conn);
    }
}
?>

login.php (User Login):-

<?php
include 'db.php';

if ($_SERVER["REQUEST_METHOD"] == "POST") {
    $email = $_POST['email'];
    $pass  = $_POST['password'];

    $sql = "SELECT * FROM users WHERE email='$email'";
    $result = mysqli_query($conn, $sql);

    if ($row = mysqli_fetch_assoc($result)) {
        if (password_verify($pass, $row['password'])) {
            echo "Login Successful! Welcome " . $row['name'];
        } else {
            echo "Invalid password!";
        }
    } else {
        echo "No user found!";
    }
}
?>

Create Table

Once RDS is created, connect via MySQL client or phpMyAdmin:
CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(50),
  email VARCHAR(100) UNIQUE,
  password VARCHAR(255)
);
